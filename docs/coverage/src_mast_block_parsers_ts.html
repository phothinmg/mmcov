	parseCodeBlock( 
		lines: SourceLine[], 
		startIndex: number,
	): { node: AstToken; nextIndex: number } { 
		const startLine = lines[startIndex] as SourceLine; 
		const language = startLine.text.startsWith("```") 
			? startLine.text.slice(3).trim() 
			: ""; 
		const content = []; 
		let index = startIndex + 1; 

		while (index < lines.length && lines[index]?.text !== "```") { 
			content.push(lines[index]?.text); 
			index++; 
		} 

		if (index < lines.length && lines[index]?.text === "```") { 
			index++; 
		} 

		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine; 

		return { 
			node: { 
				type: "codeBlock", 
				lang: language || "text", 
				raw: content.join("\n"), 
				position: mastHelpers.createPosition( 
					startLine, 
					0, 
					endLine, 
					endLine.text.length,
				), 
			}, 
			nextIndex: index, 
		}; 
	}, 
	parseLatexBlock( 
		lines: SourceLine[], 
		startIndex: number, 
	): { node: AstToken; nextIndex: number } { 
		const content = []; 
		let index = startIndex; 

		while (index < lines.length) { 
			content.push(lines[index]?.text); 

			if (index !== startIndex && lines[index]?.text.endsWith("$")) { 
				index++; 
				break; 
			} 

			if ( 
				index === startIndex && 
				(lines[index] as SourceLine).text.length > 2 && 
				lines[index]?.text.endsWith("$") 
			) { 
				index++; 
				break; 
			} 

			index++; 
		} 

		const startLine = lines[startIndex] as SourceLine; 
		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine; 

		return { 
			node: { 
				type: "latex", 
				value: content.join("\n"), 
				position: mastHelpers.createPosition( 
					startLine, 
					0, 
					endLine, 
					endLine.text.length, 
				), 
			}, 
			nextIndex: index, 
		}; 
	}, 
	parseQuote( 
		lines: SourceLine[], 
		startIndex: number, 
		parseBlockFunc: ParseBlock, 
	): { node: AstToken; nextIndex: number } { 
		const quoteLines: SourceLine[] = []; 
		let index = startIndex; 

		while (index < lines.length) { 
			const line = lines[index]; 

			if (!line?.text.trim()) { 
				break; 
			} 

			if (!blockHelpers.isQuoteStart(line.text)) { 
				break; 
			} 

			const quotePrefixLength = line.text.startsWith("> ") ? 2 : 1; 
			quoteLines.push(mastHelpers.shiftSourceLine(line, quotePrefixLength)); 
			index++; 
		} 

		const { nodes } = parseBlockFunc(quoteLines, 0); 
		const startLine = lines[startIndex] as SourceLine;
		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine; 

		return { 
			node: { 
				type: "blockquote", 
				children: nodes as AstTokens, 
				position: mastHelpers.createPosition( 
					startLine, 
					0, 
					endLine, 
					endLine.text.length, 
				), 
			}, 
			nextIndex: index, 
		};

	parseTable( 
		lines: SourceLine[], 
		startIndex: number, 
		inlineParserExtensions?: InlineParsers, 
	): { node: AstToken; nextIndex: number } { 
		const headerLine = lines[startIndex] as SourceLine; 
		const header = blockHelpers.splitTableRow(headerLine.text); 
		const align = blockHelpers.parseTableAlignment( 
			(lines[startIndex + 1] as SourceLine).text, 
		); 
		const rows: string[][] = []; 
		let index = startIndex + 2; 

		while (index < lines.length) { 
			const row = blockHelpers.splitTableRow((lines[index] as SourceLine).text); 

			if (!row || row.length !== header?.length) { 
				break; 
			} 

			rows.push(row); 
			index++; 
		} 

		const endLine = lines[Math.max(startIndex + 1, index - 1)] as SourceLine; 

		return { 
			node: { 
				type: "table", 
				align, 
				header: header?.map((cell) =>
					parseInline(cell, inlineParserExtensions), 
				), 
				rows: rows.map((row) =>
					row.map((cell) => parseInline(cell, inlineParserExtensions)), 
				), 
				position: mastHelpers.createPosition(
					headerLine, 
					0, 
					endLine, 
					endLine.text.length, 
				), 
			}, 
			nextIndex: index, 
		}; 
	}, 
	parseList( 
		lines: SourceLine[], 
		startIndex: number, 
		options: { task: boolean; ordered: boolean }, 
		parseBlockFunc: ParseBlock, 
		inlineParserExtensions?: InlineParsers, 
	): { node: AstToken; nextIndex: number } { 
		const baseIndent = blockHelpers.getIndentationWidth( 
			(lines[startIndex] as SourceLine).text, 
		); 
		const items: ListItemToken[] = []; 
		let index = startIndex; 

		while (index < lines.length) { 
			const line = lines[index] as SourceLine; 

			if (!line?.text.trim()) { 
				break; 
			} 

			const indentation = blockHelpers.getIndentationWidth(line.text); 

			if (indentation !== baseIndent) {
				break;
			}

			const markerMatch = blockHelpers.matchListMarker(
				line.text.slice(baseIndent),
				options,
			);

			if (!markerMatch) {
				break;
			}

			const item: ListItemToken = {
				type: "listItem",
				checked: options.task ? markerMatch.checked : null,
				children: [],
				position: mastHelpers.createPosition(line, 0, line, line.text.length),
			};
			const inlineText = markerMatch.content;
			if (inlineText) {
				const contentStartIndex = line.text.indexOf(inlineText, baseIndent);
				const ns = parseInline(inlineText, inlineParserExtensions, {
					text: inlineText,
					getPoint(charIndex: number) {
						return mastHelpers.pointAt(line, contentStartIndex + charIndex);
					},
				});
				for (const n of ns) {
					item.children.push(n);
				}
			}

			index++;

			if (index < lines.length) {
				const nested = blockHelpers.collectIndentedLines(
					lines,
					index,
					baseIndent + 2,
				);

				if (nested.lines.length) {
					const { nodes } = parseBlockFunc(nested.lines, 0);
					item.children.push(...(nodes as AstTokens));
					index = nested.nextIndex;
					item.position = mastHelpers.createPosition(
						line,
						0,
						lines[index - 1] as SourceLine,
						(lines[index - 1] as SourceLine).text.length,
					);
				}
			}

			items.push(item);
		}

		const startLine = lines[startIndex] as SourceLine;
		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine;

		return {
			node: {
				type: "list",
				ordered: options.ordered,
				task: options.task,
				items,
				position: mastHelpers.createPosition(
					startLine,
					0,
					endLine,
					endLine.text.length,
				),
			},
			nextIndex: index,
		};
	},
	parseHtml(
		lines: SourceLine[],
		startIndex: number,
	): { node: AstToken; nextIndex: number } {
		const htmlLines = [];
		let index = startIndex;
		let currentTag = null;
		const startLine = lines[startIndex] as SourceLine;
		const startsWithComment = startLine.text.trim().startsWith("<!--");

		if (startsWithComment) {
			while (index < lines.length) {
				const line = lines[index] as SourceLine;
				htmlLines.push(line.text);
				index++;

				if (line.text.includes("-->")) {
					break;
				}
			}

			const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine;

			return {
				node: {
					type: "comment",
					value: htmlLines.join("\n"),
					position: mastHelpers.createPosition(
						startLine,
						0,
						endLine,
						endLine.text.length,
					),
				},
				nextIndex: index,
			};
		}

		if (/^<([A-Za-z][^\s/>]*)[^>]*>$/.test(startLine.text.trim())) {
			const match = /^<([A-Za-z][^\s/>]*)[^>]*>$/.exec(startLine.text.trim());

			if (match && !startLine.text.trim().startsWith("</")) {
				currentTag = match[1];
			}
		}

		while (index < lines.length) {
			const line = lines[index];
			const trimmedLine = line?.text.trim();

			if (!trimmedLine) {
				break;
			}

			htmlLines.push(line?.text);

			if (currentTag && trimmedLine === `</${currentTag}>`) {
				index++;
				break;
			}

			if (!currentTag && trimmedLine.endsWith(">")) {
				index++;
				if (index >= lines.length || !lines[index]?.text.trim()) {
					break;
				}

				if (
					!blockHelpers.isHtmlContinuationLine(
						(lines[index] as SourceLine).text,
					)
				) {
					break;
				}

				continue;
			}

			index++;
		}

		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine;

		return {
			node: {
				type: "html",
				value: htmlLines.join("\n"),
				position: mastHelpers.createPosition(
					startLine,
					0,
					endLine,
					endLine.text.length,
				),
			},
			nextIndex: index,
		};
	},
	parseParagraph(
		lines: SourceLine[],
		startIndex: number,
		inlineParserExtensions?: InlineParsers,
	): { node: AstToken; nextIndex: number } {
		const paragraphLines: SourceLine[] = [];
		let index = startIndex;

		while (index < lines.length) {
			const line = lines[index];
			const nextLine = index + 1 < lines.length ? lines[index + 1] : undefined;

			if (!line?.text.trim()) {
				break;
			}

			if (
				paragraphLines.length &&
				blockHelpers.isBlockStart(line.text, nextLine?.text ?? "")
			) {
				break;
			}

			paragraphLines.push(line);
			index++;
		}
		const inlineSource = mastHelpers.createInlineSource(paragraphLines);
		const startLine = lines[startIndex] as SourceLine;
		const endLine = lines[Math.max(startIndex, index - 1)] as SourceLine;

		return {
			node: {
				type: "paragraph",
				children: parseInline(
					inlineSource.text,
					inlineParserExtensions,
					inlineSource,
				),
				position: mastHelpers.createPosition(
					startLine,
					0,
					endLine,
					endLine.text.length,
				),
			},
			nextIndex: index,
		};
	},
